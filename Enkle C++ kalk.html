#include <iostream>
using namespace std;
int main()
{
int valg = 0;
float tall1;
float tall2;
int heltall1;
int heltall2;
while (valg != 7) {
cout << "Du kann velge mellom 7 valg: \n" << "Valg 1 = +. \n" << "Valg 2 = -. \n" << "Valg 3 = *. \n" << "Valg 4 = /. \n" << "Valg 5 heltalldeling. \n" << "Valg 6 = %\n" << "Valg 7 avslutte programmet" << endl << "Skriv inn valget ditt : \n";
cin >> valg;
switch (valg) {
case(1):
cout << "Skriv inn to tallenen som u vil regne sammen med en mellomromstegn: \n";
cin >> tall1 >> tall2;
cout << "Svaret blir: " << tall1 + tall2 << endl;
break;
case(2):
cout << "Skriv inn to tallenen som u vil regne sammen med en mellomromstegn: \n";
cin >> tall1 >> tall2;
cout << "Svaret blir: " << tall1 - tall2 << endl;
break;
case(3):
cout << "Skriv inn to tallenen som u vil regne sammen med en mellomromstegn: \n";
cin >> tall1 >> tall2;
cout << "Svaret blir: " << tall1 * tall2 << endl;
break;
case(4):
cout << "Skriv inn to tallenen som u vil regne sammen med en mellomromstegn: \n";
cin >> tall1 >> tall2;
if (tall2 == 0) {
cout << "Du kan ikke skrive 0 som andre tall!\n";
}
else {
cout << "Svaret blir: " << tall1 / tall2 << endl;
}
break;
case(5):
cout << "Skriv inn to heltallenen som u vil regne sammen med en mellomromstegn: \n";
cin >> heltall1 >> heltall2;
if (heltall2 == 0) {
cout << "Du kan ikke skrive 0 som andre tall!\n";
}
else {
cout << "Svaret blir: " << heltall1 / heltall2 << endl;
}
break;
case(6):
cout << "Skriv inn to heltallenen som u vil regne sammen med en mellomromstegn: \n";
cin >> heltall1 >> heltall2;
if (heltall2 == 0) {
cout << "Du kan ikke skrive 0 som andre tall!\n";
}
else {
cout << "Svaret blir: " << heltall1 % heltall2 << endl;
}
break;
case(7):
cout << "programmet avsluttes";
break;
default:
cout << "Ugyldig valg, prøv igjen.\n";
break;
}
}
return 0;
system("pause>0");
}
